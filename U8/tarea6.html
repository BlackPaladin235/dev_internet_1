const libro = { titulo: "Cien años de soledad", autor: "Gabriel García Márquez", fechaPublicacion: "1967" }; // Mostrar el objeto en la consola console.log(libro); // Mostrar el objeto en el navegador const mostrarLibro = document.createElement("pre"); mostrarLibro.textContent = JSON.stringify(libro, null, 2); document.body.appendChild(mostrarLibro); // Crear la variable con la fecha actual const date = new Date().toLocaleDateString(); // Mostrar la fecha en la consola console.log(date); // Mostrar la fecha en el navegador const mostrarFecha = document.createElement("p"); mostrarFecha.textContent = `Fecha actual: ${date}`; document.body.appendChild(mostrarFecha);
